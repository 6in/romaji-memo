   ```typescript
   import { create } from 'zustand';

   export interface DocumentParagraph {
     id: string;
     input: string;
     output: string;
     styleId: string;
     createdAt: number;
   }

   interface DocumentStore {
     paragraphs: DocumentParagraph[];
     appendParagraph: (input: string, output: string, styleId: string) => void;
     removeParagraph: (id: string) => void;
     clearDocument: () => void;
     getExportContent: (format: 'md' | 'txt') => string;
   }
   ```
   - `appendParagraph`: `crypto.randomUUID()` で id 生成、配列末尾に追加
   - `removeParagraph`: filter で削除
   - `clearDocument`: 空配列にリセット
   - `getExportContent('md')`: 各段落を `## 入力\n\n{input}\n\n## 変換\n\n{output}` 形式で `\n\n---\n\n` で結合
   - `getExportContent('txt')`: 各段落の `output` のみを `\n\n` で結合

3. **src/lib/tauri.ts** に `exportDocument` 関数を追加:
   ```typescript
   import { save } from '@tauri-apps/plugin-dialog';
   import { writeTextFile } from '@tauri-apps/plugin-fs';

   export async function exportDocument(content: string, format: 'md' | 'txt'): Promise<boolean> {
     const path = await save({
       filters: [{ name: format === 'md' ? 'Markdown' : 'Text', extensions: [format] }],
       defaultPath: `document.${format}`,
     });
     if (!path) return false;
     await writeTextFile(path, content);
     return true;
   }
   ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-03-05 | Tampering | exportDocument | mitigate | `save()` ダイアログで選択したパスのみ書き込み可能。Tauri の capabilities スコープにより任意パスへの書き込みを防止 |

| T-03-07 | Denial of Service | appendParagraph | accept | メモリ内の段落配列は上限なし。実用上、数百段落を超える使用は想定外。将来的に上限を追加可能 |
