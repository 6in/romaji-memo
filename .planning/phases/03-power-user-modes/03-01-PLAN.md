   ```typescript
   import { getCurrentWindow } from '@tauri-apps/api/window';
   import { LogicalSize } from '@tauri-apps/api/dpi';

   const MINI_HEIGHT = 120; // TitleBar(32px) + textarea(~78px) + padding

   export async function enterMiniMode(): Promise<{ width: number; height: number }> {
     const win = getCurrentWindow();
     const size = await win.outerSize();
     // tauri.conf.json の minHeight 制約を一時解除してから setSize のみで縮小
     await win.setMinSize(null);
     await win.setSize(new LogicalSize(size.width, MINI_HEIGHT));
     return { width: size.width, height: size.height };
   }

   export async function exitMiniMode(savedSize: { width: number; height: number }) {
     const win = getCurrentWindow();
     await win.setSize(new LogicalSize(savedSize.width, savedSize.height));
     // minHeight 制約を tauri.conf.json の値に復元
     await win.setMinSize(new LogicalSize(320, 400));
   }
   ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| clipboard -> input | 外部アプリからのクリップボードテキストがアプリ入力欄に自動挿入される |
| app -> filesystem | エクスポート時にユーザー選択パスへファイル書き込み (Plan 02 で実装) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Denial of Service | useClipboardWatch | mitigate | onTextUpdate コールバックに 300ms デバウンスを入れ、大量クリップボード更新による CPU 負荷を防止 |
| T-03-02 | Information Disclosure | useClipboardWatch | mitigate | 監視中インジケーターを TitleBar に表示し、ユーザーが監視状態を常に認識できるようにする |
| T-03-03 | Tampering | clipboard self-copy | mitigate | コピー操作後 1000ms は監視コールバックを無視する clipboardIgnoreUntil メカニズム |
| T-03-04 | Elevation of Privilege | capabilities scope | accept | dialog:allow-save + fs:allow-write-text-file は save() ダイアログで選択されたパスのみ書き込み可。Tauri 組み込みスコープで制限 |
</threat_model>
