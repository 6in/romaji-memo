2. Create `src-tauri/src/commands/convert.rs`:
   ```rust
   #[tauri::command]
   pub async fn convert(
       state: tauri::State<'_, crate::state::AppState>,
       input: String,
       style_id: String,
       provider_id: String,
   ) -> Result<ConvertResult, String> {
       // 1. Get provider from RwLock map (drop lock before await)
       let provider = {
           let providers = state.providers.read().await;
           providers.get(&provider_id).cloned()
               .ok_or_else(|| format!("Provider '{}' not found", provider_id))?
       };

       // 2. Build system prompt from style_id (Rust-side, per anti-pattern rule)
       let system = crate::providers::prompts::build_system_prompt(&style_id);

       // 3. Create CompletionRequest
       let req = CompletionRequest {
           system,
           user_message: input.clone(),
           model: provider_model.clone(), // from provider config
           max_tokens: 1024,
       };

       // 4. Call provider (async — no DB lock held)
       let response = provider.complete(req).await.map_err(|e| e.to_string())?;

       // 5. Parse AI response JSON (defensive: strip markdown fences)
       let output = crate::providers::extract_json(&response.content)
           .map_err(|e| e.to_string())?;

       // 6. Save to history via spawn_blocking (NEVER block async runtime)
       let db = state.db.clone();
       let input_clone = input.clone();
       let style_clone = style_id.clone();
       let provider_clone = provider_id.clone();
       let model_clone = response.model.clone();
       let output_clone = output.clone();
       let history_id = tokio::task::spawn_blocking(move || {
           let conn = db.blocking_lock();
           crate::db::conversions::insert_conversion(
               &conn,
               &input_clone,
               &output_clone.converted,
               &style_clone,
               Some(&output_clone.intent),
               if output_clone.typo.is_empty() { None } else { Some(&output_clone.typo) },
               &provider_clone,
               &model_clone,
           )
       }).await.map_err(|e| e.to_string())?.map_err(|e| e.to_string())?;

       // 7. Return result to frontend
       Ok(ConvertResult {
           converted: output.converted,
           intent: output.intent,
           typo: output.typo,
           history_id,
       })
   }
   ```
   Define `ConvertResult` struct with `#[derive(serde::Serialize)]`: `{ converted: String, intent: String, typo: String, history_id: i64 }`.

5. Create `src-tauri/src/commands/window.rs`:
   ```rust
   #[tauri::command]
   pub async fn toggle_always_on_top(window: tauri::Window) -> Result<bool, String> {
       let current = window.is_always_on_top().map_err(|e| e.to_string())?;
       let new_state = !current;
       window.set_always_on_top(new_state).map_err(|e| e.to_string())?;
       Ok(new_state)
   }

   #[tauri::command]
   pub async fn save_window_state(
       state: tauri::State<'_, crate::state::AppState>,
       x: i32, y: i32, width: u32, height: u32,
   ) -> Result<(), String> {
       let json = serde_json::json!({ "x": x, "y": y, "width": width, "height": height });
       let db = state.db.clone();
       tokio::task::spawn_blocking(move || {
           let conn = db.blocking_lock();
           crate::db::settings::set_setting(&conn, "window_state", &json.to_string())
       }).await.map_err(|e| e.to_string())?.map_err(|e| e.to_string())
   }

   #[tauri::command]
   pub async fn get_window_state(
       state: tauri::State<'_, crate::state::AppState>,
   ) -> Result<Option<String>, String> {
       let db = state.db.clone();
       tokio::task::spawn_blocking(move || {
           let conn = db.blocking_lock();
           crate::db::settings::get_setting(&conn, "window_state")
       }).await.map_err(|e| e.to_string())?.map_err(|e| e.to_string())
   }
   ```
