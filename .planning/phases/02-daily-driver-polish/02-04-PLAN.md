function SortableBufferItem({ item, onRemove }: { item: BufferItem; onRemove: (id: string) => void }) {
  const { attributes, listeners, setNodeRef, transform, transition, isDragging } = useSortable({ id: item.id });
  const style = {
    transform: CSS.Transform.toString(transform),
    transition,
    opacity: isDragging ? 0.7,
  };
  return (
    <div ref={setNodeRef} style={style} className={`group flex items-center px-3 py-2 h-9 ${isDragging ? 'shadow-lg scale-[1.02]' : ''}`}>
      <button {...attributes} {...listeners} className="cursor-grab active:cursor-grabbing mr-2 text-muted-foreground">
        <GripVertical size={14} />
      </button>
      <span className="text-xs text-foreground truncate flex-1">{item.text}</span>
      <button onClick={() => onRemove(item.id)} className="p-0.5 opacity-0 group-hover:opacity-100 transition-opacity">
        <X size={12} className="text-muted-foreground hover:text-destructive" />
      </button>
    </div>
  );
}
```

const handleDragEnd = (event: DragEndEvent) => {
  const { active, over } = event;
  if (!over || active.id === over.id) return;
  const oldIndex = items.findIndex((i) => i.id === active.id);
  const newIndex = items.findIndex((i) => i.id === over.id);
  reorder(oldIndex, newIndex);
};
```
